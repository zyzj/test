SELECT CONTAINER_NUMBER,
       VESSEL_CODE,
       DECODE(OPERATION_TYPE, 'L', OPER_OUT_VOYAGE, OPER_IN_VOYAGE) VOYAGE,
       '' BL_SO_NO,
       CONTAINER_SIZE_TYPE,
       OWNER_CODE,
       CONTAINER_SERVICE_TYPE,
       OPERATION_TYPE,
       OPERATION_TIME
FROM (
SELECT RPT04.CONTAINER_NUMBER,
       BER01.VESSEL_CODE,
       BER01.OPER_IN_VOYAGE,
       BER01.OPER_OUT_VOYAGE,
       '' BL_SO_NO,
       RPT04.CONTAINER_SIZE||RPT04.CONTAINER_TYPE CONTAINER_SIZE_TYPE,
       RPT04.OWNER_CODE,
       RPT04.CONTAINER_SERVICE_TYPE,
       RPT04.OPERATION_TYPE,
       RPT04.OPERATION_TIME
FROM RPT04_VESSEL_REPORT_TB RPT04,
     BER01_VESSEL_SCHEDULE_TB BER01 
WHERE CONTAINER_NUMBER = 'HDMU2690036'
ORDER BY RPT04.OPERATION_TIME DESC
)
WHERE ROWNUM = 1
